int main(){
    char a[25], b[25];
    scanf("%s %s", a, b);
    int n = strlen(a), m = strlen(b);
    int count_a[200] = {0};
    for (int i = 0; i < n; i++) {
        count_a[a[i]]++;
    }
    int count_b[200] = {0};
    for (int i = 0; i < m; i++) {
        if (count_a[b[i]] > 0 && count_b[b[i]] == 0) {
            printf("%c ", b[i]);
            count_b[b[i]]++;
        }
    }
}
